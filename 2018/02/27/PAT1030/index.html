PAT1030. Travel Plan (30)

题目

代码


#include <iostream>
#include <vector>

using namespace std;

const int inf = 99999999;

vector<vector<int>> getPathFromStartToThisPoint(int point, vector<vector<int>> &pre) {
    if (pre[point][0] == -1) {
        return vector<vector<int>>(1, vector<int>(1, point));
    }
    else {
        vector<vector<int>> childpath;
        vector<vector<int>> resultpath;
        for (int i = 0; i < pre[point].size(); ++i) {
            childpath = getPathFromStartToThisPoint(pre[point][i], pre);
            for (int j = 0; j < childpath.size(); ++j) {
                childpath[j].push_back(point);
            }
            if (resultpath.empty()) {
                resultpath = childpath;
            }
            else {
                resultpath.insert(resultpath.end(), childpath.begin(), childpath.end());
            }
        }
        return resultpath;
    }
}


int main() {
    int cityNum, highwayNum, start, destination;
    cin >> cityNum >> highwayNum >> start >> destination;
    vector<vector<int>> Graph(cityNum, vector<int>(cityNum, inf));

    vector<vector<int>> Cost(cityNum, vector<int>(cityNum, inf));
    vector<bool> visited(cityNum, false);

    for (int i = 0; i < highwayNum; ++i) {
        int c1, c2, dist, cost;
        cin >> c1 >> c2 >> dist >> cost;
        Graph[c1][c2] = dist;
        Graph[c2][c1] = dist;
        Cost[c1][c2] = cost;
        Cost[c2][c1] = cost;
    }

    vector<vector<int>> pre(cityNum);
    pre[start].push_back(-1);

    vector<int> shortestDist(cityNum, inf);
    shortestDist[start] = 0;
    //dijkstra
    for (int i = 0; i < cityNum; ++i) {
        // choose point
        int min = inf;
        int index = -1;
        for (int j = 0; j < cityNum; ++j) {
            if (!visited[j] && min > shortestDist[j]) {
                min = shortestDist[j];
                index = j;
            }
        }
        visited[index] = true;
        // relax
        for (int j = 0; j < cityNum; ++j) {
            if (shortestDist[j] > shortestDist[index] + Graph[index][j]) {
                shortestDist[j] = shortestDist[index] + Graph[index][j];
                pre[j].clear();
                pre[j].push_back(index);
            }
            else if (shortestDist[j] == shortestDist[index] + Graph[index][j]) {
                pre[j].push_back(index);
            }
        }
    }

    vector<vector<int>> paths = getPathFromStartToThisPoint(destination, pre);

    int minCost = inf;
    int minIndex = -1;
    for (int j = 0; j < paths.size(); ++j) {
        int costtmp = 0;
        for (int i = 0; i < paths[j].size() - 1; ++i) {
            costtmp += Cost[paths[j][i]][paths[j][i + 1]];
        }
        if (minCost > costtmp) {
            minCost = costtmp;
            minIndex = j;
        }
    }
    for (auto elem : paths[minIndex]) {
        cout << elem << " ";
    }
    cout << shortestDist[destination] << " " << minCost;

    return 0;
}

总结

这道题和PAT1018非常类似，甚至比1018要简单，原理还是一样：找出所有最短路径，再按照题中要求求得最符合要求的哪一条路径即可。

图数据结构（二维矩阵）中同一点之间的距离不再初始化为0，而是初始化为无穷大，防止在之后添加前向点的时候把自身这个点也添加进去，稍微改进了一下dijkstra部分的代码，以及回溯路径的代码，但实际上差不多。
